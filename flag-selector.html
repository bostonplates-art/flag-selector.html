    <option value="White" data-price="15">White (£15)</option>

    <option value="Blue" data-price="20">Blue (£20)</option>

  </select>

  

  <div id="flag"></div>

  <div id="price">£0</div>

  <button id="buyNow">Buy Now</button>


  <script>

    const flagSelect = document.getElementById("flagSelect");

    const flagDiv = document.getElementById("flag");

    const priceDiv = document.getElementById("price");

    const buyBtn = document.getElementById("buyNow");


    let selected = { flag: "", price: 0 };


    flagSelect.addEventListener("change", (e) => {

      const option = e.target.selectedOptions[0];

      selected.flag = option.value;

      selected.price = option.dataset.price || 0;


      flagDiv.style.background = option.value.toLowerCase();

      priceDiv.textContent = "£" + selected.price;

    });


    buyBtn.addEventListener("click", () => {

      if (!selected.flag) {

        alert("Please select a flag first");

        return;

      }

      // Send to parent Wix site

      window.parent.postMessage(selected, "*");

    });

  </script>
